
## cau hinh tren sw distributed

```
enable
configure terminal
hostname SW_AP
vtp domain cisco.com
vtp password cisco
vtp mode server
vtp version 1

vlan 10
name manager
vlan 20
name nha1
vlan 30
name nha2
int vlan 10
ip addr 192.168.250.1 255.255.255.0
no shut
exit
int vlan 20
ip addr 192.168.251.1 255.255.255.0
no shut
exit
int vlan 30
ip addr 192.168.252.1 255.255.255.0
no shut
exit

do show vlan br

int e0/0
switchport trunk encapsulation dot1q 
switchport mode trunk

```
## cau hinh tren sw access 
```
enable
configure terminal
hostname SW_AP
enable password cisco

int e0/0
switchport trunk encapsulation dot1q 
switchport mode trunk

do sh int trunk

vtp domain cisco.com
vtp password cisco
vtp mode client

int vlan 10
no shut
int vlan 20
no shut
int vlan 30
no shut
do sh ip int br

```
