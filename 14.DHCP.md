

- Cấu hình trên Server 1. IP :  192.168.122.200
```
yum -y install dhcp
cp -np  /etc/dhcp/dhcpd.conf
cat <<EOF > /etc/dhcp/dhcpd.conf
# create new
# specify domain name
option domain-name "8.8.8.8";
# specify name server's hostname or IP address
option domain-name-servers 192.168.122.200;
# default lease time
default-lease-time 600;
# max lease time
max-lease-time 7200;
# this DHCP server to be declared valid
authoritative;

failover peer "failover-partner" {
     primary;
     address 192.168.122.200;
     port 647;
     peer address 192.168.122.201;
     peer port 647;
     max-response-delay 60;
     max-unacked-updates 10;
     mclt 3600;
     split 128;
     load balance max seconds 3;
}
# specify network address and subnet mask
subnet 192.168.122.0 netmask 255.255.255.0 {
    option domain-name-servers 1.1.1.1;
    option routers 192.168.122.1;
    pool {
        failover peer "failover-partner";
        range 192.168.122.220 192.168.122.230;
    }
}



EOF
firewall-cmd --add-service=dhcp --permanent 
firewall-cmd --add-port=647/tcp --permanent

firewall-cmd --reload  

systemctl restart dhcpd 
systemctl enable dhcpd 
```

- Cấu hình trên Client 
```
## xoá ip động 
dhclient -r 

## nhận ip động 

dhclient eth0 



## chu y device 
cat <<EOF > etc/sysconfig/network-scripts/ifcfg-eth0
DEVICE=eth0
TYPE=Ethernet
NAME=eth0
BOOTPROTO=dhcp
ONBOOT=yes
EOF
